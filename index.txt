    <script>
        // --- 1. CONFIGURATION ---

            storageBucket: "dilistname.firebasestorage.app",
            messagingSenderId: "897983357871",
            appId: "1:897983357871:web:42a046bc9fb3e0543dc55a",
            measurementId: "G-NQ798D9J6K"
        };
        firebase.initializeApp(firebaseConfig);
        const db = firebase.database();

        // YOUR CLOUDINARY CONFIG (REPLACE THESE!)
        const CLOUDINARY_URL = "https://api.cloudinary.com/v1_1/YOUR_CLOUD_NAME_HERE/image/upload";
        const CLOUDINARY_UPLOAD_PRESET = "YOUR_UNSIGNED_PRESET_HERE"; 

        // --- 2. VARIABLES ---
        let currentStudentId = null;
        const video = document.getElementById('videoFeed');
        const canvas = document.getElementById('canvas');
        const defaultImage = "https://via.placeholder.com/150?text=n/a"; // n/a placeholder

        // --- 3. SEARCH FUNCTION ---
        function searchStudent() {
            const id = document.getElementById('searchInput').value.trim();
            if(!id) return alert("Please enter an ID");

            // UI Reset
            document.getElementById('loading').classList.remove('hidden');
            document.getElementById('profileCard').classList.add('hidden');
            document.getElementById('notFound').classList.add('hidden');

            // Fetch from Firebase: students/{id}
            db.ref('students/' + id).once('value').then((snapshot) => {
                document.getElementById('loading').classList.add('hidden');
                
                if (snapshot.exists()) {
                    const data = snapshot.val();
                    currentStudentId = id; // Save ID for later update
                    renderProfile(data);
                } else {
                    document.getElementById('notFound').classList.remove('hidden');
                }
            }).catch(error => {
                console.error(error);
                alert("Error connecting to database");
                document.getElementById('loading').classList.add('hidden');
            });
        }

        // --- 4. RENDER UI ---
        function renderProfile(data) {
            // Map Data (Khmer Fields)
            document.getElementById('dispName').innerText = data['ឈ្មោះ'] || "Unknown";
            document.getElementById('dispID').innerText = data['អត្តលេខ'] || currentStudentId;
            document.getElementById('dispSkill').innerText = data['ជំនាញ'] || "General";

            const imgUrl = data['រូបថត'];
            const imgEl = document.getElementById('displayImage');
            const btnCam = document.getElementById('btnOpenCamera');
            const overlay = document.getElementById('cameraOverlay');

            // Logic: If image exists, show it and hide camera button. If not, show n/a and show button.
            if (imgUrl && imgUrl !== "n/a" && imgUrl !== "") {
                imgEl.src = imgUrl;
                btnCam.classList.add('hidden'); // Hide button
                overlay.classList.add('hidden'); // Hide hover effect
            } else {
                imgEl.src = defaultImage;
                btnCam.classList.remove('hidden'); // Show Capture Button
                overlay.classList.remove('hidden'); // Enable hover effect
            }

            document.getElementById('profileCard').classList.remove('hidden');
        }

        // --- 5. CAMERA LOGIC ---
        function openCameraModal() {
            document.getElementById('cameraModal').classList.remove('hidden');
            navigator.mediaDevices.getUserMedia({ video: { facingMode: "environment" } })
                .then(stream => {
                    video.srcObject = stream;
                })
                .catch(err => {
                    alert("Camera access denied or not available");
                    closeCamera();
                });
        }

        function closeCamera() {
            const stream = video.srcObject;
            if (stream) {
                const tracks = stream.getTracks();
                tracks.forEach(track => track.stop());
            }
            document.getElementById('cameraModal').classList.add('hidden');
        }

        // --- 6. CAPTURE & UPLOAD (The Magic) ---
        function capturePhoto() {
            // 1. Draw video frame to canvas
            const context = canvas.getContext('2d');
            canvas.width = video.videoWidth;
            canvas.height = video.videoHeight;
            context.drawImage(video, 0, 0, canvas.width, canvas.height);

            // 2. Convert to Base64
            const dataUrl = canvas.toDataURL('image/jpeg');

            // 3. Stop Camera & Show Loading State
            closeCamera();
            document.getElementById('loading').classList.remove('hidden');
            document.getElementById('profileCard').classList.add('hidden');

            // 4. Upload to Cloudinary
            uploadToCloudinary(dataUrl);
        }

        function uploadToCloudinary(base64Image) {
            const formData = new FormData();
            formData.append('file', base64Image);
            formData.append('upload_preset', CLOUDINARY_UPLOAD_PRESET);

            fetch(CLOUDINARY_URL, {
                method: 'POST',
                body: formData
            })
            .then(response => response.json())
            .then(data => {
                if(data.secure_url) {
                    updateFirebase(data.secure_url);
                } else {
                    alert("Upload failed");
                    document.getElementById('loading').classList.add('hidden');
                }
            })
            .catch(err => {
                console.error(err);
                alert("Error uploading image");
                document.getElementById('loading').classList.add('hidden');
            });
        }

        // --- 7. UPDATE FIREBASE ---
        function updateFirebase(newImageUrl) {
            db.ref('students/' + currentStudentId).update({
                'រូបថត': newImageUrl
            }).then(() => {
                // Success! Refresh the view
                alert("Photo saved successfully!");
                searchStudent(); // Reload data
            }).catch(error => {
                alert("Failed to update database");
            });
        }
